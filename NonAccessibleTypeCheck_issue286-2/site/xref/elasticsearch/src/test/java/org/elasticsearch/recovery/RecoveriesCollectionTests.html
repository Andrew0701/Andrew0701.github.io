59      @Test
60      public void testLastAccessTimeUpdate() throws Exception {
61          createIndex();
62          final RecoveriesCollection collection = new RecoveriesCollection(logger, getInstanceFromNode(ThreadPool.class));
63          final long recoveryId = startRecovery(collection);
64          try (RecoveriesCollection.StatusRef status = collection.getStatus(recoveryId)) {
65              final long lastSeenTime = status.status().lastAccessTime();
66              assertBusy(new Runnable() {
67                  @Override
68                  public void run() {
69                      try (RecoveriesCollection.StatusRef currentStatus = collection.getStatus(recoveryId)) {
70                          assertThat("access time failed to update", lastSeenTime, lessThan(currentStatus.status().lastAccessTime()));
71                      }
72                  }
73              });
74          } finally {
75              collection.cancelRecovery(recoveryId, "life");
76          }
77      }

79      @Test
80      public void testRecoveryTimeout() throws InterruptedException {
81          createIndex();
82          final RecoveriesCollection collection = new RecoveriesCollection(logger, getInstanceFromNode(ThreadPool.class));
83          final AtomicBoolean failed = new AtomicBoolean();
84          final CountDownLatch latch = new CountDownLatch(1);
85          final long recoveryId = startRecovery(collection, new RecoveryTarget.RecoveryListener() {
86              @Override
87              public void onRecoveryDone(RecoveryState state) {
88                  latch.countDown();
89              }
90  
91              @Override
92              public void onRecoveryFailure(RecoveryState state, RecoveryFailedException e, boolean sendShardFailure) {
93                  failed.set(true);
94                  latch.countDown();
95              }
96          }, TimeValue.timeValueMillis(100));
97          try {
98              latch.await(30, TimeUnit.SECONDS);
99              assertTrue("recovery failed to timeout", failed.get());
100         } finally {
101             collection.cancelRecovery(recoveryId, "meh");
102         }
103 
104     }

106     @Test
107     public void testRecoveryCancellationNoPredicate() throws Exception {
108         createIndex();
109         final RecoveriesCollection collection = new RecoveriesCollection(logger, getInstanceFromNode(ThreadPool.class));
110         final long recoveryId = startRecovery(collection);
111         final long recoveryId2 = startRecovery(collection);
112         try (RecoveriesCollection.StatusRef statusRef = collection.getStatus(recoveryId)) {
113             ShardId shardId = statusRef.status().shardId();
114             assertTrue("failed to cancel recoveries", collection.cancelRecoveriesForShard(shardId, "test"));
115             assertThat("all recoveries should be cancelled", collection.size(), equalTo(0));
116         } finally {
117             collection.cancelRecovery(recoveryId, "meh");
118             collection.cancelRecovery(recoveryId2, "meh");
119         }
120     }

122     @Test
123     public void testRecoveryCancellationPredicate() throws Exception {
124         createIndex();
125         final RecoveriesCollection collection = new RecoveriesCollection(logger, getInstanceFromNode(ThreadPool.class));
126         final long recoveryId = startRecovery(collection);
127         final long recoveryId2 = startRecovery(collection);
128         final ArrayList<AutoCloseable> toClose = new ArrayList<>();
129         try {
130             RecoveriesCollection.StatusRef statusRef = collection.getStatus(recoveryId);
131             toClose.add(statusRef);
132             ShardId shardId = statusRef.status().shardId();
133             assertFalse("should not have cancelled recoveries", collection.cancelRecoveriesForShard(shardId, "test", Predicates.<RecoveryStatus>alwaysFalse()));
134             final Predicate<RecoveryStatus> shouldCancel = new Predicate<RecoveryStatus>() {
135                 @Override
136                 public boolean apply(RecoveryStatus status) {
137                     return status.recoveryId() == recoveryId;
138                 }
139             };
140             assertTrue("failed to cancel recoveries", collection.cancelRecoveriesForShard(shardId, "test", shouldCancel));
141             assertThat("we should still have on recovery", collection.size(), equalTo(1));
142             statusRef = collection.getStatus(recoveryId);
143             toClose.add(statusRef);
144             assertNull("recovery should have been deleted", statusRef);
145             statusRef = collection.getStatus(recoveryId2);
146             toClose.add(statusRef);
147             assertNotNull("recovery should NOT have been deleted", statusRef);
148 
149         } finally {
150             // TODO: do we want a lucene IOUtils version of this?
151             for (AutoCloseable closeable : toClose) {
152                 if (closeable != null) {
153                     closeable.close();
154                 }
155             }
156             collection.cancelRecovery(recoveryId, "meh");
157             collection.cancelRecovery(recoveryId2, "meh");
158         }
159     }

161     protected void createIndex() {
162         createIndex("test",
163                 ImmutableSettings.builder()
164                         .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1, IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 0)
165                         .build());
166         ensureGreen();
167     }

170     long startRecovery(RecoveriesCollection collection) {
171         return startRecovery(collection, listener, TimeValue.timeValueMinutes(60));
172     }
173 
174     long startRecovery(RecoveriesCollection collection, RecoveryTarget.RecoveryListener listener, TimeValue timeValue) {
175         IndicesService indexServices = getInstanceFromNode(IndicesService.class);
176         IndexShard indexShard = indexServices.indexServiceSafe("test").shard(0);
177         final DiscoveryNode sourceNode = new DiscoveryNode("id", DummyTransportAddress.INSTANCE, Version.CURRENT);
178         return collection.startRecovery(indexShard, sourceNode, listener, timeValue);
179     }
