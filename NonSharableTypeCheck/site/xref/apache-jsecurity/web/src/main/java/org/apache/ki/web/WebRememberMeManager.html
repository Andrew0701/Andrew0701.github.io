34  /**
35   * Remembers a Subject's identity by using a {@link WebAttribute WebAttribute} instance to retain
36   * the identity value between web requests.
37   *
38   * <p>This class's default <code>WebAttribute</code> instance is a {@link org.apache.ki.web.attr.CookieAttribute CookieAttribute}, storing
39   * the Subject's {@link org.apache.ki.subject.Subject#getPrincipals principals} in a <code>Cookie</code>.  Note that
40   * because this class subclasses the <code>AbstractRememberMeManager</code> which already provides serialization and
41   * encryption logic, this class utilizes both for added security before setting the cookie value.</p>
42   *
43   * <p>This class also contains &quot;passthrough&quot; JavaBeans-compatible getters/setters for the underlying
44   * <code>CookieAttribute</code>'s properties to make configuration easier.</p>
45   *
46   * <p>Note however as a basic sanity check, these passthrough methods will first assert that the underlying
47   * {@link #getIdentityAttribute identityAttribute} is actually a {@link CookieAttribute CookieAttribute}.  If it
48   * is not, an {@link IllegalStateException} will be thrown.  Because the default instance of this class <em>is</em>
49   * already <code>CookieAttribute</code>, you would only ever experience the exception if you explicitly
50   * override the internal instance with a different type and accidentally call one of these JavaBeans passthrough
51   * methods.</p>
52   *
53   * <p>Just be aware of this if you manually override the {@link #getIdentityAttribute identityAttribute} property
54   * to be an instance of something <em>other</em> than a <code>CookieAttribute</code>.</p>
55   *
56   * @author Les Hazlewood
57   * @author Luis Arias
58   * @since 0.9
59   */
60  public class WebRememberMeManager extends AbstractRememberMeManager {

66      /**
67       * The default name of the underlying rememberMe cookie which is <code>rememberMe</code>.
68       */
69      public static final String DEFAULT_REMEMBER_ME_COOKIE_NAME = "rememberMe";
70  
71      protected WebAttribute<String> identityAttribute = null;
72  
73      public WebRememberMeManager() {
74          CookieAttribute<String> attr = new CookieAttribute<String>(DEFAULT_REMEMBER_ME_COOKIE_NAME);
75          attr.setCheckRequestParams(false);
76          //Peter (Apache Ki developer) said that Jetty didn't like the CookieAttribute.INDEFINITE value
77          // (Tomcat was ok with it), so just default to a few years for now.  If anyone doesn't visit a site in 3 years
78          // after last login, I doubt any Ki users would mind their end-users to be forced to log in. - LAH.
79          attr.setMaxAge(CookieAttribute.ONE_YEAR * 3);
80          this.identityAttribute = attr;
81      }
82  
83      public WebAttribute<String> getIdentityAttribute() {
84          return identityAttribute;
85      }
86  
87      public void setIdentityAttribute(WebAttribute<String> identityAttribute) {
88          this.identityAttribute = identityAttribute;
89      }
90  
91      protected void assertCookieAttribute() {
92          if (!(this.identityAttribute instanceof CookieAttribute)) {
93              String msg = "Attempting to access a Cookie property, but the underlying " +
94                      WebAttribute.class.getName() + " instance is not a " +
95                      CookieAttribute.class.getName() + " instance.  This is expected if you " +
96                      "are accessing or modifying a cookie property.";
97              throw new IllegalStateException(msg);
98          }
99      }

201     protected void rememberSerializedIdentity(byte[] serialized) {
202         ServletRequest request = WebUtils.getRequiredServletRequest();
203         ServletResponse response = WebUtils.getRequiredServletResponse();
204         //base 64 encode it and store as a cookie:
205         String base64 = Base64.encodeToString(serialized);
206         getIdentityAttribute().storeValue(base64, request, response);
207     }
208 
209     protected boolean isIdentityRemoved() {
210         ServletRequest request = WebUtils.getServletRequest();
211         if ( request != null ) {
212             Boolean removed = (Boolean)request.getAttribute(KiHttpServletRequest.IDENTITY_REMOVED_KEY);
213             return removed != null && removed;
214         }
215         return false;
216     }
217     protected byte[] getSerializedRememberedIdentity() {
218         if ( isIdentityRemoved() ) {
219             return null;
220         }
221 
222         ServletRequest request = WebUtils.getRequiredServletRequest();
223         ServletResponse response = WebUtils.getRequiredServletResponse();
224         String base64 = getIdentityAttribute().retrieveValue(request, response);
225         if (base64 != null) {
226             base64 = ensurePadding(base64);
227             if (log.isTraceEnabled()) {
228                 log.trace("Acquired Base64 encoded identity [" + base64 + "]");
229             }
230             byte[] decoded = Base64.decode(base64);
231             if (log.isTraceEnabled()) {
232                 log.trace("Base64 decoded byte array length: " + (decoded != null ? decoded.length : 0) + " bytes.");
233             }
234             return decoded;
235         } else {
236             //no cookie set - new site visitor?
237             return null;
238         }
239     }

241     /**
242      * Sometimes a user agent will send the rememberMe cookie value without padding,
243      * most likely because {@code =} is a separator in the cookie header.  Contributed
244      * by Luis Arias.
245      *
246      * @param base64 the base64 encoded String that may need to be padded
247      * @return the base64 String padded if necessary.
248      */
249     private String ensurePadding(String base64) {
250         int length = base64.length();
251         if (length % 4 != 0) {
252             StringBuffer sb = new StringBuffer(base64);
253             for (int i = 0; i < length % 4; ++i) {
254                 sb.append('=');
255             }
256             base64 = sb.toString();
257         }
258         return base64;
259     }

262     protected void forgetIdentity() {
263         ServletRequest request = WebUtils.getRequiredServletRequest();
264         ServletResponse response = WebUtils.getRequiredServletResponse();
265         getIdentityAttribute().removeValue(request, response);
266     }
