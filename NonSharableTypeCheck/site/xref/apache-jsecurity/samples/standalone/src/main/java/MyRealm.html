20  import org.apache.ki.authc.*;
21  import org.apache.ki.authz.AuthorizationInfo;
22  import org.apache.ki.realm.AuthorizingRealm;
23  import org.apache.ki.subject.PrincipalCollection;
24  
25  import java.util.Collection;
26  
27  /**
28   * @author Les Hazlewood
29   * @since Aug 28, 2008 5:13:06 PM
30   */
31  public class MyRealm extends AuthorizingRealm {
32  
33      public MyRealm() {
34      }
35  
36      /**
37       * Simulates a call to an underlying data store - in a 'real' application, this call would communicate with
38       * an underlying data store via an EIS API (JDBC, JPA, Hibernate, etc).
39       * <p/>
40       * Note that when implementing your own realm, there is no need to check against a password (or other credentials)
41       * in this method. The {@link org.apache.ki.realm.AuthenticatingRealm AuthenticatingRealm} superclass will do
42       * that automatically via the use of a configured
43       * {@link org.apache.ki.authc.credential.CredentialsMatcher CredentialsMatcher} (see this example's corresponding
44       * <code>ki.ini</code> file to see a configured credentials matcher).
45       * <p/>
46       * All that is required is that the account information include directly the credentials found in the EIS.
47       *
48       * @param username the username for the account data to retrieve
49       * @return the Account information corresponding to the specified username:
50       */
51      protected SimpleAccount getAccount(String username) {
52          //just create a dummy.  A real app would construct one based on EIS access.
53          SimpleAccount account = new SimpleAccount(username, "sha256EncodedPasswordFromDatabase", getName());
54          //simulate some roles and permissions:
55          account.addRole("user");
56          account.addRole("admin");
57          //most applications would assign permissions to Roles instead of users directly because this is much more
58          //flexible (it is easier to configure roles and then change role-to-user assignments than it is to maintain
59          // permissions for each user).
60          // But these next lines assign permissions directly to this trivial account object just for simulation's sake:
61          account.addStringPermission("blogEntry:edit"); //this user is allowed to 'edit' _any_ blogEntry
62          //fine-grained instance level permission:
63          account.addStringPermission("printer:print:laserjet2000"); //allowed to 'print' to the 'printer' identified
64          //by the id 'laserjet2000'
65  
66          return account;
67      }
68  
69      protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken token) throws AuthenticationException {
70          //we can safely cast to a UsernamePasswordToken here, because this class 'supports' UsernamePasswordToken
71          //objects.  See the Realm.supports() method if your application will use a different type of token.
72          UsernamePasswordToken upToken = (UsernamePasswordToken) token;
73          return getAccount(upToken.getUsername());
74      }
75  
76      protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principals) {
77          //simulate a call to an underlying data store:
78  
79          //get only the principals that this realm cares about:
80          Collection thisRealmPrincipals = principals.fromRealm(getName());
81  
82          //note that the return value of 'getName()' here is whatever you specify it to be in ki.ini
83          //in this case, the ki.ini file calls it 'myRealm'.  All realms must have a unique name.
84  
85          //Since this realm supplied these principals from the doGetAuthenticationInfo method above when the
86          //user logged-in and that method in this simple example has only one principal - a username - we can safely
87          //assume the only element in this collection is that username.
88          String username = (String) thisRealmPrincipals.iterator().next();
89  
90          //call the underlying EIS for the account data:
91          return getAccount(username);
92      }
93  }
