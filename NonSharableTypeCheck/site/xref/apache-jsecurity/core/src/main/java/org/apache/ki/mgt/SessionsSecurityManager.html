39  /**
40   * Ki support of a {@link SecurityManager} class hierarchy that delegates all
41   * {@link org.apache.ki.session.Session session} operations to a wrapped {@link org.apache.ki.session.mgt.SessionManager SessionManager}
42   * instance.  That is, this class implements the methods in the
43   * {@link SessionManager SessionManager} interface, but in reality, those methods are merely passthrough calls to
44   * the underlying 'real' {@code SessionManager} instance.
45   * <p/>
46   * The remaining {@code SecurityManager} methods not implemented by this class or its parents are left to be
47   * implemented by subclasses.
48   * <p/>
49   * In keeping with the other classes in this hierarchy and Ki's desire to minimize configuration whenever
50   * possible, suitable default instances for all dependencies will be created upon instantiation.
51   *
52   * @author Les Hazlewood
53   * @since 0.9
54   */
55  public abstract class SessionsSecurityManager extends AuthorizingSecurityManager implements SessionListenerRegistrar {
56  
57      /**
58       * The internal delegate <code>SessionManager</code> used by this security manager that manages all the
59       * application's {@link Session Session}s.
60       */
61      private SessionManager sessionManager;
62  
63      /**
64       * Default no-arg constructor, internally creates a suitable default {@link SessionManager SessionManager} delegate
65       * instance.
66       */
67      public SessionsSecurityManager() {
68          super();
69          this.sessionManager = new DefaultSessionManager();
70          applyCacheManagerToSessionManager();
71      }
72  
73      /**
74       * Sets the underlying delegate {@link SessionManager} instance that will be used to support this implementation's
75       * <tt>SessionManager</tt> method calls.
76       * <p/>
77       * <p>This <tt>SecurityManager</tt> implementation does not provide logic to support the inherited
78       * <tt>SessionManager</tt> interface, but instead delegates these calls to an internal
79       * <tt>SessionManager</tt> instance.
80       * <p/>
81       * <p>If a <tt>SessionManager</tt> instance is not set, a default one will be automatically created and
82       * initialized appropriately for the the existing runtime environment.
83       *
84       * @param sessionManager delegate instance to use to support this manager's <tt>SessionManager</tt> method calls.
85       */
86      public void setSessionManager(SessionManager sessionManager) {
87          this.sessionManager = sessionManager;
88          afterSessionManagerSet();
89      }
90  
91      protected void afterSessionManagerSet() {
92          applyCacheManagerToSessionManager();
93      }
94  
95      /**
96       * Returns this security manager's internal delegate {@link SessionManager SessionManager}.
97       *
98       * @return this security manager's internal delegate {@link SessionManager SessionManager}.
99       * @see #setSessionManager(org.apache.ki.session.mgt.SessionManager) setSessionManager
100      */
101     public SessionManager getSessionManager() {
102         return this.sessionManager;
103     }
104 
105     /**
106      * Calls {@link org.apache.ki.mgt.AuthorizingSecurityManager#afterCacheManagerSet() super.afterCacheManagerSet()} and then immediately calls
107      * {@link #applyCacheManagerToSessionManager() applyCacheManagerToSessionManager()} to ensure the
108      * <code>CacheManager</code> is applied to the SessionManager as necessary.
109      */
110     protected void afterCacheManagerSet() {
111         super.afterCacheManagerSet();
112         applyCacheManagerToSessionManager();
113     }
114 
115     /**
116      * Ensures the internal delegate <code>SessionManager</code> is injected with the newly set
117      * {@link #setCacheManager CacheManager} so it may use it for its internal caching needs.
118      * <p/>
119      * Note:  This implementation only injects the CacheManager into the SessionManager if the SessionManager
120      * instance implements the {@link CacheManagerAware CacheManagerAware} interface.
121      */
122     protected void applyCacheManagerToSessionManager() {
123         if (this.sessionManager instanceof CacheManagerAware) {
124             ((CacheManagerAware) this.sessionManager).setCacheManager(getCacheManager());
125         }
126     }
127 
128     /**
129      * This is a convenience method that allows registration of SessionListeners with the underlying delegate
130      * SessionManager at startup.
131      * <p/>
132      * <p>This is more convenient than having to configure your own SessionManager instance, inject the listeners on
133      * it, and then set that SessionManager instance as an attribute of this class.  Instead, you can just rely
134      * on the <tt>SecurityManager</tt> to apply these <tt>SessionListener</tt>s on your behalf.
135      * <p/>
136      * <p>One notice however: The underlying SessionManager delegate must implement the
137      * {@link SessionListenerRegistrar SessionListenerRegistrar} interface in order for these listeners to
138      * be applied.  If it does not implement this interface, it is considered a configuration error and an exception
139      * will be thrown.
140      *
141      * @param sessionListeners the <tt>SessionListener</tt>s to register with the underlying delegate
142      *                         <tt>SessionManager</tt> at startup.
143      */
144     public void setSessionListeners(Collection<SessionListener> sessionListeners) {
145         assertSessionListenerSupport();
146         ((SessionListenerRegistrar) this.sessionManager).setSessionListeners(sessionListeners);
147     }
148 
149     private void assertSessionManager(Class<? extends SessionManager> requiredType) {
150         if (this.sessionManager == null) {
151             throw new IllegalStateException("SessionManager is null - cannot configure property!");
152         }
153         if (!(requiredType.isInstance(this.sessionManager))) {
154             String msg = "Property configuration failed.  The target property is only configurable when the " +
155                 "underlying SessionManager instance is a part of the " +
156                 "[" + requiredType.getName() + "] class hierarchy.  " +
157                 "The current SessionManager is of type [" + this.sessionManager.getClass().getName() + "].  " +
158                 "This might occur for example if you're trying to set the validation interval or auto session " +
159                 "creation in a servlet container-backed session environment ('http' session mode).  If that is " +
160                 "the case however, that property is only useful when using 'ki' session mode and using " +
161                 "Ki enterprise sessions which do not rely on a servlet container.";
162             throw new IllegalStateException(msg);
163         }
164     }

166     /**
167      * Passthrough configuration property to the underlying {@link AbstractSessionManager AbstractSessionManager}
168      * instance.  Please read the
169      * {@link org.apache.ki.session.mgt.AbstractSessionManager#getGlobalSessionTimeout() AbstractSessionManager.getGlobalSessionTimeout()}
170      * for more.
171      *
172      * @return the time in milliseconds that any {@link Session Session} may remain idle before expiring.
173      * @throws IllegalStateException if the underlying {@code SessionManager} instance is not a subclass of
174      *                               {@link AbstractSessionManager AbstractSessionManager}.
175      * @see org.apache.ki.session.mgt.AbstractSessionManager#getGlobalSessionTimeout()
176      */
177     public long getGlobalSessionTimeout() {
178         assertSessionManager(AbstractSessionManager.class);
179         return ((AbstractSessionManager) this.sessionManager).getGlobalSessionTimeout();
180     }
181 
182     /**
183      * Passthrough configuration property to the underlying {@link AbstractSessionManager AbstractSessionManager}
184      * instance.  Please read the
185      * {@link org.apache.ki.session.mgt.AbstractSessionManager#setGlobalSessionTimeout(long) AbstractSessionManager.setGlobalSessionTimeout(long)}
186      * for more.
187      *
188      * @param globalSessionTimeout the time in milliseconds that any {@link Session Session} may remain idle before expiring.
189      * @throws IllegalStateException if the underlying {@code SessionManager} instance is not a subclass of
190      *                               {@link org.apache.ki.session.mgt.AbstractSessionManager AbstractSessionManager}.
191      * @see org.apache.ki.session.mgt.AbstractSessionManager#setGlobalSessionTimeout(long)
192      */
193     public void setGlobalSessionTimeout(long globalSessionTimeout) {
194         assertSessionManager(AbstractSessionManager.class);
195         ((AbstractSessionManager) this.sessionManager).setGlobalSessionTimeout(globalSessionTimeout);
196     }

198     /**
199      * Passthrough configuration property to the wrapped {@link org.apache.ki.session.mgt.AbstractValidatingSessionManager} - if it should
200      * automatically create a new session when an invalid session is referenced.  The default value unless
201      * overridden by this method is <code>true</code> for developer convenience and to match what most people are
202      * accustomed based on years of servlet container behavior.
203      * <p/>
204      * When true (the default), the wrapped {@link AbstractValidatingSessionManager} implementation throws an
205      * {@link org.apache.ki.session.ReplacedSessionException ReplacedSessionException} to the caller whenever a new
206      * session is created so the caller can receive the new session ID and react accordingly for future
207      * {@link SessionManager SessionManager} method invocations.
208      *
209      * @param autoCreate if the wrapped {@link AbstractValidatingSessionManager} should automatically create a new
210      *                   session when an invalid session is referenced
211      * @see org.apache.ki.session.mgt.AbstractValidatingSessionManager#setAutoCreateWhenInvalid(boolean)
212      */
213     public void setAutoCreateSessionAfterInvalidation(boolean autoCreate) {
214         assertSessionManager(AbstractValidatingSessionManager.class);
215         ((AbstractValidatingSessionManager) this.sessionManager).setAutoCreateWhenInvalid(autoCreate);
216     }
217 
218     /**
219      * Passthrough configuration property that returns <code>true</code> if the wrapped
220      * {@link org.apache.ki.session.mgt.AbstractValidatingSessionManager AbstractValidatingSessionManager} should automatically create a
221      * new session when an invalid session is referenced, <code>false</code> otherwise.  Unless overridden by the
222      * {@link #setAutoCreateSessionAfterInvalidation(boolean)} method, the default value is <code>true</code> for
223      * developer convenience and to match what most people are accustomed based on years of servlet container behavior.
224      * <p/>
225      * When true (the default), the wrapped {@link org.apache.ki.session.mgt.AbstractValidatingSessionManager AbstractValidatingSessionManager}
226      * implementation throws an {@link org.apache.ki.session.ReplacedSessionException ReplacedSessionException} to
227      * the caller whenever a new session is created so the caller can receive the new session ID and react accordingly
228      * for future {@link SessionManager SessionManager} method invocations.
229      *
230      * @return <code>true</code> if this session manager should automatically create a new session when an invalid
231      *         session is referenced, <code>false</code> otherwise.
232      * @see org.apache.ki.session.mgt.AbstractValidatingSessionManager#isAutoCreateWhenInvalid()
233      */
234     public boolean isAutoCreateSessionAfterInvalidation() {
235         assertSessionManager(AbstractValidatingSessionManager.class);
236         return ((AbstractValidatingSessionManager) this.sessionManager).isAutoCreateWhenInvalid();
237     }
238 
239     /**
240      * Ensures the internal SessionManager instance is an <code>instanceof</code>
241      * {@link org.apache.ki.session.SessionListenerRegistrar SessionListenerRegistrar} to ensure that any
242      * listeners attempting to be registered can actually do so with the internal delegate instance.
243      *
244      * @throws IllegalStateException if the internal delegate SessionManager instance does not implement the
245      *                               <code>SessionListenerRegistrar</code> interface.
246      */
247     private void assertSessionListenerSupport() throws IllegalStateException {
248         if (!(this.sessionManager instanceof SessionListenerRegistrar)) {
249             String msg = "SessionListener registration failed:  The underlying SessionManager instance of " +
250                 "type [" + sessionManager.getClass().getName() + "] does not implement the " +
251                 SessionListenerRegistrar.class.getName() + " interface and therefore cannot support " +
252                 "session notifications.";
253             throw new IllegalStateException(msg);
254         }
255     }
256 
257     /**
258      * Asserts the internal delegate <code>SessionManager</code> instance
259      * {@link #assertSessionListenerSupport() supports session listener registration} and then
260      * {@link SessionListenerRegistrar#add adds} the listener to the
261      * delegate instance.
262      *
263      * @param listener the <code>SessionListener</code> to register for session events.
264      */
265     public void add(SessionListener listener) {
266         assertSessionListenerSupport();
267         ((SessionListenerRegistrar) this.sessionManager).add(listener);
268     }
269 
270     /**
271      * Removes the specified listener from receiving session events from the internal delegate
272      * {@link org.apache.ki.session.mgt.SessionManager} instance.
273      *
274      * @param listener the listener to remove that no longer wishes to be notified of session events.
275      * @return <code>true</code> if the listener was removed from the internal delegate <code>SessionManager</code>
276      *         instance, <code>false</code> otherwise.
277      */
278     public boolean remove(SessionListener listener) {
279         return (this.sessionManager instanceof SessionListenerRegistrar) &&
280             ((SessionListenerRegistrar) this.sessionManager).remove(listener);
281     }
