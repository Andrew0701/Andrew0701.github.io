25  import org.apache.ki.cache.Cache;
26  import org.apache.ki.cache.CacheManager;

33  /**
34   * An CachingSessionDAO is a SessionDAO that provides a transparent caching layer between the components that
35   * use it and the underlying EIS (Enterprise Information System) for enhanced performance.
36   *
37   * <p>This implementation caches all active sessions in a cache created by a
38   * {@link org.apache.ki.cache.CacheManager}.  All <tt>SessionDAO</tt> methods are implemented by this class to employ
39   * caching behavior and delegates the actual EIS operations to respective do* methods to be implemented by
40   * subclasses (doCreate, doRead, etc).
41   *
42   * @author Les Hazlewood
43   * @since 0.2
44   */
45  public abstract class CachingSessionDAO implements SessionDAO, CacheManagerAware {
46  
47      /**
48       * The default active sessions cache name, equal to <code>ki-activeSessionCache</code>.
49       */
50      public static final String ACTIVE_SESSION_CACHE_NAME = "ki-activeSessionCache";
51  
52      /**
53       * The CacheManager to use to acquire the Session cache.
54       */
55      private CacheManager cacheManager;
56  
57      /**
58       * The Cache instance responsible for caching Sessions.
59       */
60      private Cache activeSessions;
61  
62      /**
63       * The name of the session cache, defaults to {@link #ACTIVE_SESSION_CACHE_NAME}.
64       */
65      private String activeSessionsCacheName = ACTIVE_SESSION_CACHE_NAME;
66  
67      /**
68       * Default no-arg constructor.
69       */
70      public CachingSessionDAO() {
71      }
72  
73      /**
74       * Sets the cacheManager to use for constructing the session cache.
75       *
76       * @param cacheManager the manager to use for constructing the session cache.
77       */
78      public void setCacheManager(CacheManager cacheManager) {
79          this.cacheManager = cacheManager;
80          //force cache reload:
81          this.activeSessions = null;
82      }
83  
84      /**
85       * Returns the CacheManager used by the implementation that creates the activeSessions Cache.
86       *
87       * @return the CacheManager used by the implementation that creates the activeSessions Cache.
88       */
89      public CacheManager getCacheManager() {
90          return cacheManager;
91      }
92  
93      /**
94       * Returns the name of the actives sessions cache to be returned by the <code>CacheManager</code>.  Unless
95       * overridden by {@link #setActiveSessionsCacheName(String)}, defaults to {@link #ACTIVE_SESSION_CACHE_NAME}.
96       * @return the name of the active sessions cache.
97       */
98      public String getActiveSessionsCacheName() {
99          return activeSessionsCacheName;
100     }
101 
102     /**
103      * Sets the name of the active sessions cache to be returned by the <code>CacheManager</code>.  Defaults to
104      * {@link #ACTIVE_SESSION_CACHE_NAME}.
105      * @param activeSessionsCacheName the name of the active sessions cache to be returned by the <code>CacheManager</code>. 
106      */
107     public void setActiveSessionsCacheName(String activeSessionsCacheName) {
108         this.activeSessionsCacheName = activeSessionsCacheName;
109     }
110 
111     /**
112      * Returns the cache instance to use for storing active sessions.
113      * @return the cache instance to use for storing active sessions.
114      */
115     public Cache getActiveSessionsCache() {
116         return this.activeSessions;
117     }
118 
119     /**
120      * Returns the active sessions cache, but if that cache instance is null, first lazily creates the cache instance
121      * via the {@link #createActiveSessionsCache()} method and then returns the instance.
122      * <p/>
123      * Note that this method will only return a non-null value code if the <code>CacheManager</code> has been set.  If
124      * not set, there will be no cache.
125      * 
126      * @return the active sessions cache instance.
127      */
128     protected Cache getActiveSessionsCacheLazy() {
129         if (this.activeSessions == null) {
130             this.activeSessions = createActiveSessionsCache();
131         }
132         return this.activeSessions;
133     }
134 
135     /**
136      * Sets the cache instance to use for storing active sessions.
137      * @param cache the cache instance to use for storing active sessions.
138      */
139     public void setActiveSessionsCache(Cache cache) {
140         this.activeSessions = cache;
141     }
142 
143     /**
144      * Creates a cache instance used to store active sessions.  Creation is done by first
145      * {@link #getCacheManager() acquiring} the <code>CacheManager</code>.  If the cache manager is not null, the
146      * cache returned is that resulting from the following call:
147      * <pre>       String name = {@link #getActiveSessionsCacheName() getActiveSessionsCacheName()};
148      * cacheManager.getCache(name);</pre>
149      * @return a cache instance used to store active sessions, or <em>null</code> if the <code>CacheManager</code> has
150      * not been set.
151      */
152     protected Cache createActiveSessionsCache() {
153         Cache cache = null;
154         CacheManager mgr = getCacheManager();
155         if (mgr != null) {
156             String name = getActiveSessionsCacheName();
157             cache = mgr.getCache(name);
158         }
159         return cache;
160     }
161 
162     /**
163      * Creates the session by delegating EIS creation to subclasses via the {@link #doCreate} method, and then
164      * caches the session.
165      *
166      * @param session Session object to create in the EIS and then cache.
167      */
168     public Serializable create(Session session) {
169         Serializable sessionId = doCreate(session);
170         verifySessionId(sessionId);
171         cacheValidSession(session, sessionId);
172         return sessionId;
173     }
174 
175     /**
176      * Returns the cached session with the corresponding <code>sessionId</code> or <code>null</code> if there is
177      * no session cached under that id (or if there is no Cache).
178      *
179      * @param sessionId the id of the cached session to acquire.
180      * @return the cached session with the corresponding <code>sessionId</code>, or <code>null</code> if the session
181      * does not exist or is not cached.
182      */
183     protected Session getCachedSession(Serializable sessionId) {
184         Session cached = null;
185         if (sessionId != null) {
186             Cache cache = getActiveSessionsCacheLazy();
187             if (cache != null) {
188                 cached = getCachedSession(sessionId, cache);
189             }
190         }
191         return cached;
192     }
193 
194     /**
195      * Returns the Session with the specified id from the specified cache.  This method simply calls
196      * <code>cache.get(sessionId)</code> and can be overridden by subclasses for custom acquisition behavior.
197      * @param sessionId the id of the session to acquire.
198      * @param cache the cache to acquire the session from
199      * @return the cached session, or <code>null</code> if the session wasn't in the cache.
200      */
201     protected Session getCachedSession(Serializable sessionId, Cache cache) {
202         return (Session) cache.get(sessionId);
203     }
204 
205     /**
206      * Caches the specified session under the key <code>sessionId</code>.  If the Session is an instance of
207      * {@link org.apache.ki.session.mgt.ValidatingSession ValidatingSession}, it will only be cached if the
208      * session is {@link org.apache.ki.session.mgt.ValidatingSession#isValid() valid}.
209      *
210      * @param session the session to cache
211      * @param sessionId the id of the session, to be used as the cache key.
212      */
213     protected void cacheValidSession(Session session, Serializable sessionId) {
214         if (session == null || sessionId == null) {
215             return;
216         }
217 
218         Cache cache = getActiveSessionsCacheLazy();
219         if (cache == null) {
220             return;
221         }
222 
223         if (session instanceof ValidatingSession && !((ValidatingSession) session).isValid()) {
224             uncache(session);
225         } else {
226             cache(session, sessionId, cache);
227         }
228     }
229 
230     /**
231      * Caches the specified session in the given cache under the key of <code>sessionId</code>.  This implementation
232      * simply calls <code>cache.put(sessionId, session)</code> and can be overridden for custom behavior.
233      * 
234      * @param session the session to cache
235      * @param sessionId the id of the session, expected to be the cache key.
236      * @param cache the cache to store the session
237      */
238     protected void cache(Session session, Serializable sessionId, Cache cache) {
239         cache.put(sessionId, session);
240     }
241 
242     /**
243      * Ensures the sessionId returned from the subclass implementation of {@link #doCreate} is not null and not
244      * already in use.
245      *
246      * @param sessionId session id returned from the subclass implementation of {@link #doCreate}
247      */
248     protected void verifySessionId(Serializable sessionId) {
249         if (sessionId == null) {
250             String msg = "sessionId returned from doCreate implementation is null.  Please verify the implementation.";
251             throw new IllegalStateException(msg);
252         }
253         ensureUncached(sessionId);
254     }
255 
256     /**
257      * Ensures that there is no cache entry already in place for a session with id of <tt>sessionId</tt>.  Used by
258      * the {@link #verifySessionId} implementation.
259      *
260      * @param sessionId the session id to check for non-existence in the cache.
261      */
262     protected void ensureUncached(Serializable sessionId) {
263         Cache cache = getActiveSessionsCacheLazy();
264         if (cache != null && cache.get(sessionId) != null) {
265             String msg = "There is an existing session already created with session id [" +
266                     sessionId + "].  Session ID's must be unique.";
267             throw new IllegalArgumentException(msg);
268         }
269     }
270 
271     /**
272      * Subclass hook to actually persist the given <tt>Session</tt> instance to the underlying EIS.
273      *
274      * @param session the Session instance to persist to the EIS.
275      * @return the id of the session created in the EIS (i.e. this is almost always a primary key and should be the
276      *         value returned from {@link org.apache.ki.session.Session#getId() Session.getId()}.
277      */
278     protected abstract Serializable doCreate(Session session);
279 
280     /**
281      * Retrieves the Session object from the underlying EIS identified by <tt>sessionId</tt>.
282      *
283      * <p>Upon receiving the Session object from the subclass's {@link #doReadSession} implementation, it will be
284      * cached first and then returned to the caller.
285      *
286      * @param sessionId the id of the session to retrieve from the EIS.
287      * @return the session identified by <tt>sessionId</tt> in the EIS.
288      * @throws UnknownSessionException if the id specified does not correspond to any session in the cache or EIS.
289      */
290     public Session readSession(Serializable sessionId) throws UnknownSessionException {
291 
292         Session s = getCachedSession(sessionId);
293 
294         if (s == null) {
295             s = doReadSession(sessionId);
296             if (s != null) {
297                 cacheValidSession(s, sessionId);
298             }
299         }
300 
301         if (s == null) {
302             throw new UnknownSessionException("There is no session with id [" + sessionId + "]");
303         }
304         return s;
305     }
306 
307     /**
308      * Subclass implmentation hook to actually retrieve the Session object from the underlying EIS.
309      *
310      * @param sessionId the id of the <tt>Session</tt> to retrieve.
311      * @return the Session in the EIS identified by <tt>sessionId</tt>
312      */
313     protected abstract Session doReadSession(Serializable sessionId);
314 
315     /**
316      * Updates the state of the given session to the EIS.
317      *
318      * <p>If the specified session was previously cached, and the session is now invalid,
319      * it will be removed from the cache.
320      *
321      * <p>If the specified session is not stopped or expired, and was not yet in the cache, it will be added to the
322      * cache.
323      *
324      * <p>Finally, this method calls {@link #doUpdate} for the subclass to actually push the object state to the EIS.
325      *
326      * @param session the session object to update in the EIS.
327      * @throws UnknownSessionException if no existing EIS session record exists with the
328      *                                 identifier of {@link Session#getId() session.getId()}
329      */
330     public void update(Session session) throws UnknownSessionException {
331         doUpdate(session);
332         cacheValidSession(session, session.getId());
333     }
334 
335     /**
336      * Subclass implementation hook to actually persist the <tt>Session</tt>'s state to the underlying EIS.
337      *
338      * @param session the session object whose state will be propagated to the EIS.
339      */
340     protected abstract void doUpdate(Session session);
341 
342     /**
343      * Removes the specified session from any cache and then permanently deletes the session from the EIS by
344      * delegating to {@link #doDelete}.
345      *
346      * @param session the session to remove from caches and permanently delete from the EIS.
347      */
348     public void delete(Session session) {
349         doDelete(session);
350         uncache(session);
351     }
352 
353     /**
354      * Subclass implementation hook to permanently delete the given Session from the underlying EIS.
355      *
356      * @param session the session instance to permanently delete from the EIS.
357      */
358     protected abstract void doDelete(Session session);
359 
360     /**
361      * Removes the specified Session from the cache.
362      *
363      * @param session the session to remove from the cache.
364      */
365     protected void uncache(Session session) {
366         if (session == null) {
367             return;
368         }
369         Serializable id = session.getId();
370         if (id == null) {
371             return;
372         }
373         Cache cache = getActiveSessionsCacheLazy();
374         if (cache != null) {
375             cache.remove(id);
376         }
377     }
378 
379     /**
380      * Returns all active sessions in the system.
381      *
382      * <p>This implementation merely returns the sessions found in the activeSessions cache.  Subclass implementations
383      * may wish to override this method to retrieve them in a different way, perhaps by an RDBMS query or by other
384      * means.
385      *
386      * @return the sessions found in the activeSessions cache.
387      */
388     @SuppressWarnings({"unchecked"})
389     public Collection<Session> getActiveSessions() {
390         Cache cache = getActiveSessionsCacheLazy();
391         if (cache != null) {
392             return cache.values();
393         } else {
394             return Collections.EMPTY_LIST;
395         }
396     }
