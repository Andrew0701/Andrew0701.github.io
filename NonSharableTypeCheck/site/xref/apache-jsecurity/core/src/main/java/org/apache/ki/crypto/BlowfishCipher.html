35  /**
36   * Ki's default symmetric block Cipher using the Blowfish algorithm.  As it is a symmetric Cipher, it uses the
37   * same <tt>Key</tt> to both encrypt and decrypt data.  If one is not provided via the {@link #setKey setKey} method,
38   * a default one will be used, BUT NOTE:
39   *
40   * <p>Because Ki is an open-source project, if anyone knew that you were using Ki's default
41   * <code>Key</code>, they could download/view the source, and with enough effort, reconstruct the <code>Key</code>
42   * and decode encrypted data at will.
43   *
44   * <p>Ki only really uses Ciphers to encrypt user ids and session ids, so if that information is not critical
45   * to you and you think the default key still makes things 'sufficiently difficult', then you can ignore this issue.
46   *
47   * <p>However, if you do feel this constitutes sensitive information, it is recommended that you provide your own
48   * <tt>Key</tt> via the {@link #setKey setKey} method to a Key known only to your application,
49   * guaranteeing that no third party can decrypt your data.  If you want to know how to do this, you can browse this
50   * class's source code for the {@link #generateNewKey()} method to see how we created our default.  Then you can
51   * duplicate the same in your environment and set the result on an instance of this class via the
52   * <code>setKey</code> method.
53   *
54   * @author Les Hazlewood
55   * @author Jeremy Haile
56   * @since 0.9
57   */
58  public class BlowfishCipher implements Cipher {
59  
60      /**
61       * The JDK Crypto Cipher algorithm to use for this class, equal to &quot;Blowfish&quot;.
62       */
63      private static final String ALGORITHM = "Blowfish";
64  
65      /**
66       * The JDK Crypto Transformation string to use for this class, equal to {@link #ALGORITHM ALGORITHM} + &quot;/ECB/PKCS5Padding&quot;;
67       */
68      private static final String TRANSFORMATION_STRING = ALGORITHM + "/ECB/PKCS5Padding";
69  
70      //The following KEY_BYTES String was created by running
71      //System.out.println( Base64.encode( generateNewKey().getEncoded() ) ); and copying-n-pasting the output here.
72      //You should run the same and set the resulting output as a property of this class instead of using
73      //Ki's default Key for proper security.
74      private static final byte[] KEY_BYTES = Base64.decode("jJ9Kg1BAevbvhSg3vBfwfQ==");
75      private static final Key DEFAULT_CIPHER_KEY = new SecretKeySpec(KEY_BYTES, ALGORITHM);
76  
77      /**
78       * Internal private log instance.
79       */
80      private static final Logger log = LoggerFactory.getLogger(BlowfishCipher.class);
81  
82      /**
83       * The key to use by default, can be overridden by calling {@link #setKey(java.security.Key)}.
84       */
85      private Key key = DEFAULT_CIPHER_KEY;
86  
87      /**
88       * Default no argument constructor that uses an internal default {@link #getKey() key} to use during
89       * encryption and decryption.  For propery security, you should definitely supply your own key by using the
90       * {@link #setKey(java.security.Key) setKey(Key)} method.
91       */
92      public BlowfishCipher() {
93      }
94  
95      /**
96       * Returns the default {@link Key Key} to use for symmetric encryption and decryption if one is not specified during
97       * encryption/decryption.  For truly secure applications,
98       * you should always specify your own key via the {@link #setKey(java.security.Key) setKey} method.
99       *
100      * @return the {@link Key Key} to use for symmetric encryption and decryption.
101      * @see #encrypt(byte[], byte[])
102      * @see #decrypt(byte[], byte[])
103      */
104     public Key getKey() {
105         return key;
106     }
107 
108     /**
109      * Sets the internal default {@link Key Key} to use for symmetric encryption and decryption if one is not
110      * specified during encryption/decryption.   For truly secure applications, you should always specify your own
111      * key via this method.
112      *
113      * @param key the key to use for symmetric encryption and decryption.
114      * @see #encrypt(byte[], byte[])
115      * @see #decrypt(byte[], byte[])
116      */
117     public void setKey(Key key) {
118         this.key = key;
119     }
120 
121     /**
122      * Encrypts the specified raw byte array.  If the <code>key</code> argument is null, the internal default
123      * {@link #getKey() key} will be used to encrypt the byte array.
124      */
125     public byte[] encrypt(byte[] raw, byte[] key) {
126         byte[] encrypted = crypt(raw, javax.crypto.Cipher.ENCRYPT_MODE, key);
127         if (log.isTraceEnabled()) {
128             log.trace("Incoming byte array of size " + (raw != null ? raw.length : 0) + ".  Encrypted " +
129                     "byte array is size " + (encrypted != null ? encrypted.length : 0));
130         }
131         return encrypted;
132     }
133 
134     /**
135      * Decrypts the specified already-encrypted byte array.  If the <code>key</code> argument is null, the internal default
136      * {@link #getKey() key} will be used to encrypt the byte array.
137      */
138     public byte[] decrypt(byte[] encrypted, byte[] key) {
139         if (log.isTraceEnabled()) {
140             log.trace("Attempting to decrypt incoming byte array of length " +
141                     (encrypted != null ? encrypted.length : 0));
142         }
143         return crypt(encrypted, javax.crypto.Cipher.DECRYPT_MODE, key);
144     }
145 
146     /**
147      * Returns a new {@link javax.crypto.Cipher Cipher} instance to use for encryption/decryption operations, based on
148      * the {@link #TRANSFORMATION_STRING TRANSFORMATION_STRING} constant.
149      *
150      * @return a new Cipher instance.
151      * @throws IllegalStateException if a new Cipher instance cannot be constructed based on the
152      *                               {@link #TRANSFORMATION_STRING TRANSFORMATION_STRING} constant.
153      */
154     protected javax.crypto.Cipher newCipherInstance() throws IllegalStateException {
155         try {
156             return javax.crypto.Cipher.getInstance(TRANSFORMATION_STRING);
157         } catch (Exception e) {
158             String msg = "Unable to acquire a Java JCE Cipher instance using " +
159                     javax.crypto.Cipher.class.getName() + ".getInstance( \"" + TRANSFORMATION_STRING + "\" ). " +
160                     "Blowfish under this configuration is required for the " +
161                     getClass().getName() + " instance to function.";
162             throw new IllegalStateException(msg, e);
163         }
164     }
165 
166     /**
167      * Initializes the JDK Cipher with the specified mode and key.  This is primarily a utility method to catch any
168      * potential {@link InvalidKeyException InvalidKeyException} that might arise.
169      *
170      * @param cipher the JDK Cipher to {@link javax.crypto.Cipher#init(int, java.security.Key) init}.
171      * @param mode   the Cipher mode
172      * @param key    the Cipher's Key
173      */
174     protected void init(javax.crypto.Cipher cipher, int mode, java.security.Key key) {
175         try {
176             cipher.init(mode, key);
177         } catch (InvalidKeyException e) {
178             String msg = "Unable to init cipher.";
179             throw new IllegalStateException(msg, e);
180         }
181     }
182 
183     /**
184      * Calls the {@link javax.crypto.Cipher#doFinal(byte[]) doFinal(bytes)} method, propagating any exception that
185      * might arise in an {@link IllegalStateException IllegalStateException}
186      *
187      * @param cipher the JDK Cipher to finalize (perform the actual cryption)
188      * @param bytes  the bytes to crypt
189      * @return the resulting crypted byte array.
190      */
191     protected byte[] crypt(javax.crypto.Cipher cipher, byte[] bytes) {
192         try {
193             return cipher.doFinal(bytes);
194         } catch (Exception e) {
195             String msg = "Unable to crypt bytes with cipher [" + cipher + "].";
196             throw new IllegalStateException(msg, e);
197         }
198     }
199 
200     /**
201      * Calls the {@link #init(javax.crypto.Cipher, int, java.security.Key)} and then
202      * {@link #crypt(javax.crypto.Cipher, byte[])}.  Ensures that the key is never null by using the
203      * {@link #getKey() default key} if the method argument key is <code>null</code>.
204      *
205      * @param bytes the bytes to crypt
206      * @param mode  the JDK Cipher mode
207      * @param key   the key to use to do the cryption.  If <code>null</code> the {@link #getKey() default key} will be used.
208      * @return the resulting crypted byte array
209      */
210     protected byte[] crypt(byte[] bytes, int mode, byte[] key) {
211         javax.crypto.Cipher cipher = newCipherInstance();
212 
213         java.security.Key jdkKey;
214         if (key == null) {
215             jdkKey = getKey();
216         } else {
217             jdkKey = new SecretKeySpec(key, ALGORITHM);
218         }
219 
220         init(cipher, mode, jdkKey);
221         return crypt(cipher, bytes);
222     }
223 
224     /**
225      * Generates a new {@link Key Key} suitable for this Cipher by calling
226      * {@link #generateNewKey(int) generateNewKey(128)} (uses a 128 bit size by default).
227      *
228      * @return a new {@link Key Key}, 128 bits in length.
229      */
230     public static Key generateNewKey() {
231         return generateNewKey(128);
232     }
233 
234     /**
235      * Generates a new {@link Key Key} of the specified size suitable for this Cipher
236      * (based on the {@link #ALGORITHM ALGORITHM} using the JDK {@link KeyGenerator KeyGenerator}.
237      *
238      * @param keyBitSize the bit size of the key to create
239      * @return the created key suitable for use with this Cipher.
240      */
241     public static Key generateNewKey(int keyBitSize) {
242         KeyGenerator kg;
243         try {
244             kg = KeyGenerator.getInstance(ALGORITHM);
245         } catch (NoSuchAlgorithmException e) {
246             String msg = "Unable to acquire " + ALGORITHM + " algorithm.  This is required to function.";
247             throw new IllegalStateException(msg, e);
248         }
249         kg.init(keyBitSize);
250         return kg.generateKey();
251     }

253     /**
254      * Simple test main method to ensure functionality is correct.  Should really be moved to a proper test case.
255      *
256      * @param unused ignored
257      * @throws Exception if anything unexpected happens.
258      */
259     public static void main(String[] unused) throws Exception {
260 
261         Cipher cipher = new BlowfishCipher();
262 
263         String[] cleartext = new String[]{
264                 "Hello, this is a test.",
265                 "Lorem ipsum dolor sit amet, consectetur adipisicing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua."
266         };
267 
268         for (String clear : cleartext) {
269             byte[] cleartextBytes = CodecSupport.toBytes(clear);
270             System.out.println("Clear text: [" + clear + "]");
271             System.out.println("Clear text base64: [" + Base64.encodeToString(cleartextBytes) + "]");
272 
273             byte[] encrypted = cipher.encrypt(cleartextBytes, null);
274             String encryptedBase64 = Base64.encodeToString(encrypted);
275             System.out.println("Encrypted base64: [" + encryptedBase64 + "]");
276 
277             byte[] decrypted = cipher.decrypt(Base64.decode(encryptedBase64), null);
278             String decryptedString = CodecSupport.toString(decrypted);
279 
280             System.out.println("Arrays equal? " + Arrays.equals(cleartextBytes, decrypted));
281 
282             System.out.println("Decrypted text: [" + decryptedString + "]");
283             System.out.println("Decrypted text base64: [" + Base64.encodeToString(decrypted) + "]");
284         }
285     }
