19  package org.apache.ki;
20  
21  import org.apache.ki.mgt.SecurityManager;
22  import org.apache.ki.subject.Subject;
23  import org.apache.ki.util.ThreadContext;
24  
25  
26  /**
27   * Accesses the currently accessible <tt>Subject</tt> for the calling code depending on runtime environment.
28   *
29   * @author Les Hazlewood
30   * @since 0.2
31   */
32  public abstract class SecurityUtils {
33  
34      /**
35       * ONLY used as a 'backup' in VM Singleton environments (that is, standalone environments), since the
36       * ThreadContext should always be the primary source for Subject instances when possible.
37       */
38      private static SecurityManager securityManager;
39  
40      /**
41       * Returns the currently accessible <tt>Subject</tt> available to the calling code depending on
42       * runtime environment.
43       * <p/>
44       * This method is provided as a way of obtaining a <tt>Subject</tt> without having to resort to
45       * implementation-specific methods.  It also allows the Ki team to change the underlying implementation of
46       * this method in the future depending on requirements/updates without affecting your code that uses it.
47       * <p/>
48       * <b>Implementation Note:</b> This implementation expects a
49       * {@link org.apache.ki.util.ThreadContext#getSecurityManager() thread-bound} or
50       * {@link #setSecurityManager static VM singleton} {@code SecurityManager} to be accessible to this method at
51       * runtime.  If not, an {@link IllegalStateException IllegalStateException} is thrown, indicating an incorrect
52       * application configuration.
53       *
54       * @return the currently accessible <tt>Subject</tt> accessible to the calling code.
55       *
56       * @throws IllegalStateException if no {@link org.apache.ki.mgt.SecurityManager SecurityManager} instance is available to this method
57       * at runtime, which is considered an invalid application configuration - a Subject should _always_ be available
58       * to the caller.  If you encounter an exception when calling this method, ensure that the application's
59       * {@code SecurityManager} is {@link org.apache.ki.util.ThreadContext#getSecurityManager() thread-bound} or a
60       * {@link #setSecurityManager static VM singleton} prior to calling this method.
61       */
62      public static Subject getSubject() {
63          Subject subject;
64          org.apache.ki.mgt.SecurityManager securityManager = ThreadContext.getSecurityManager();
65          if (securityManager != null) {
66              subject = securityManager.getSubject();
67          } else {
68              subject = ThreadContext.getSubject();
69              if (subject == null && SecurityUtils.securityManager != null) {
70                  //fall back to the VM singleton if one exists:
71                  subject = SecurityUtils.securityManager.getSubject();
72              }
73          }
74          if ( subject == null ) {
75              String msg = "No SecurityManager accessible to this method, either bound to the " +
76                      ThreadContext.class.getName() + " or as a vm static singleton.  See the " +
77                      SecurityUtils.class.getName() + ".getSubject() method JavaDoc for an explanation of expected " +
78                      "environment configuration.";
79              throw new IllegalStateException(msg);
80          }
81          return subject;
82      }
83  
84      /**
85       * Sets a VM (static) singleton SecurityManager, specifically for transparent use in the
86       * {@link #getSubject() getSubject()} implementation.
87       *
88       * <p><b>This method call exists mainly for framework development support.  Application developers should rarely,
89       * if ever, need to call this method.</b></p>
90       *
91       * <p>The Ki development team prefers that SecurityManager instances are non-static application singletons
92       * and <em>not</em> VM static singletons.  Application singletons that do not use static memory require some sort
93       * of application configuration framework to maintain the application-wide SecurityManager instance for you
94       * (for example, Spring or EJB3 environments) such that the object reference does not need to be static.
95       *
96       * <p>In these environments, Ki acquires Subject data based on the currently executing Thread via its own
97       * framework integration code, and this is the preferred way to use Ki.</p>
98       *
99       * <p>However in some environments, such as a standalone desktop application or Applets that do not use Spring or
100      * EJB or similar config frameworks, a VM-singleton might make more sense (although the former is still preferred).</p>
101      * In these environments, setting the SecurityManager via this method will automatically enable the
102      * {@link #getSubject() getSubject()} call to function with little configuration.</p>
103      *
104      * <p>For example, in these environments, this will work:</p>
105      *
106      * <code>DefaultSecurityManager securityManager = new {@link org.apache.ki.mgt.DefaultSecurityManager DefaultSecurityManager}();<br/>
107      * securityManager.setRealms( ... ); //one or more Realms<br/>
108      * <b>SecurityUtils.setSecurityManager( securityManager );</b></code>
109      *
110      * <p>And then anywhere in the application code, the following call will return the application's Subject:</p>
111      *
112      * <p><code>Subject currentUser = SecurityUtils.getSubject()</code></p>
113      *
114      * <p>by calling the VM static {@link org.apache.ki.mgt.SecurityManager#getSubject() securityManager.getSubject()}
115      * method.  Note that the underlying injected SecurityManager still needs to know how to acquire a Subject
116      * instance for the calling code, which might mean from static memory, or a config file, or other
117      * environment-specific means.</p>
118      *
119      * @param securityManager the securityManager instance to set as a VM static singleton.
120      */
121     public static void setSecurityManager(SecurityManager securityManager) {
122         SecurityUtils.securityManager = securityManager;
123     }
124 
125     /**
126      * Returns the VM (static) singleton SecurityManager.
127      *
128      * <p>This method is <b>only used in rare occasions</b>.  Please read the {@link #setSecurityManager setSecurityManager}
129      * JavaDoc for usage patterns.
130      *
131      * @return the VM (static) singleton SecurityManager, used only on rare occasions.
132      */
133     public static SecurityManager getSecurityManager() {
134         return SecurityUtils.securityManager;
135     }
136 }
