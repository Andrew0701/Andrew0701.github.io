19  package org.apache.ki.mgt;
20  
21  import java.util.Collection;
22  
23  import org.apache.ki.authc.AuthenticationException;
24  import org.apache.ki.authc.AuthenticationInfo;
25  import org.apache.ki.authc.AuthenticationListener;
26  import org.apache.ki.authc.AuthenticationListenerRegistrar;
27  import org.apache.ki.authc.AuthenticationToken;
28  import org.apache.ki.authc.Authenticator;
29  import org.apache.ki.authc.pam.AuthenticationStrategy;
30  import org.apache.ki.authc.pam.ModularRealmAuthenticator;
31  import org.apache.ki.util.LifecycleUtils;
32  
33  
34  /**
35   * Ki support of a {@link SecurityManager} class hierarchy that delegates all
36   * authentication operations to a wrapped {@link Authenticator Authenticator} instance.  That is, this class
37   * implements all the <tt>Authenticator</tt> methods in the {@link SecurityManager SecurityManager}
38   * interface, but in reality, those methods are merely passthrough calls to the underlying 'real'
39   * <tt>Authenticator</tt> instance.
40   *
41   * <p>All other <tt>SecurityManager</tt> (authorization, session, etc) methods are left to be implemented by subclasses.
42   *
43   * <p>In keeping with the other classes in this hierarchy and Ki's desire to minimize configuration whenever
44   * possible, suitable default instances for all dependencies are created upon instantiation.
45   *
46   * @author Les Hazlewood
47   * @since 0.9
48   */
49  public abstract class AuthenticatingSecurityManager extends RealmSecurityManager implements AuthenticationListenerRegistrar {
50  
51      /**
52       * The internal <code>Authenticator</code> delegate instance that this SecurityManager instance will use
53       * to perform all authentication operations.
54       */
55      private Authenticator authenticator;
56  
57      /**
58       * Default no-arg constructor that initializes its internal
59       * <code>authenticator</code> instance to a
60       * {@link org.apache.ki.authc.pam.ModularRealmAuthenticator ModularRealmAuthenticator}.
61       */
62      public AuthenticatingSecurityManager() {
63          super();
64          this.authenticator = new ModularRealmAuthenticator();
65      }
66  
67      /**
68       * Returns the delegate <code>Authenticator</code> instance that this SecurityManager uses to perform all
69       * authentication operations.  Unless overridden by the
70       * {@link #setAuthenticator(org.apache.ki.authc.Authenticator) setAuthenticator}, the default instance is a
71       * {@link org.apache.ki.authc.pam.ModularRealmAuthenticator ModularRealmAuthenticator}.
72       *
73       * @return the delegate <code>Authenticator</code> instance that this SecurityManager uses to perform all
74       *         authentication operations.
75       */
76      public Authenticator getAuthenticator() {
77          return authenticator;
78      }
79  
80      /**
81       * Sets the delegate <code>Authenticator</code> instance that this SecurityManager uses to perform all
82       * authentication operations.  Unless overridden by this method, the default instance is a
83       * {@link org.apache.ki.authc.pam.ModularRealmAuthenticator ModularRealmAuthenticator}.
84       *
85       * @param authenticator the delegate <code>Authenticator</code> instance that this SecurityManager will use to
86       *                      perform all authentication operations.
87       * @throws IllegalArgumentException if the argument is <code>null</code>.
88       */
89      public void setAuthenticator(Authenticator authenticator) throws IllegalArgumentException {
90          if (authenticator == null) {
91              String msg = "Authenticator argument cannot be null.";
92              throw new IllegalArgumentException(msg);
93          }
94          this.authenticator = authenticator;
95      }
96  
97      /**
98       * Sets the {@link org.apache.ki.authc.pam.AuthenticationStrategy} to use
99       * in multi-realm environments.
100      *
101      * @param strategy the <code>AuthenticationStrategy</code> to use in multi-realm environments.
102      */
103     public void setAuthenticationStrategy(AuthenticationStrategy strategy) {
104         if (!(this.authenticator instanceof ModularRealmAuthenticator)) {
105             String msg = "Configuring a AuthenticationStrategy is only applicable when the underlying " +
106                     "Authenticator implementation is a " + ModularRealmAuthenticator.class.getName() +
107                     " implementation.  This SecurityManager has been configured with an Authenticator of type " +
108                     this.authenticator.getClass().getName();
109             throw new IllegalStateException(msg);
110         }
111         ((ModularRealmAuthenticator) this.authenticator).setAuthenticationStrategy(strategy);
112     }
113 
114     /**
115      * This is a convenience method that allows registration of AuthenticationListeners with the underlying
116      * delegate Authenticator instance.
117      *
118      * <p>This is more convenient than having to configure your own Authenticator instance, inject the listeners on
119      * it, and then set that Authenticator instance as an attribute of this class.  Instead, you can just rely
120      * on the <tt>SecurityManager</tt>'s default initialization logic to create the Authenticator instance for you
121      * and then apply these <tt>AuthenticationListener</tt>s on your behalf.
122      *
123      * <p>One notice however: The underlying Authenticator delegate must implement the
124      * {@link org.apache.ki.authc.AuthenticationListenerRegistrar AuthenticationListenerRegistrar}
125      * interface in order for these listeners to be applied.  If it does not implement this interface, it is
126      * considered a configuration error and an exception will be thrown.
127      *
128      * <p>All of Ki's <tt>Authenticator</tt> implementations implement the
129      * <tt>AuthenticationListenerRegistrar</tt> interface, so you would only need
130      * to worry about an exception being thrown if you provided your own Authenticator instance and did not
131      * implement it.
132      *
133      * @param listeners the <tt>AuthenticationListener</tt>s to register with the underlying delegate
134      *                  <tt>Authenticator</tt>.
135      */
136     public void setAuthenticationListeners(Collection<AuthenticationListener> listeners) {
137         assertAuthenticatorListenerSupport();
138         if (!(this.authenticator instanceof AuthenticationListenerRegistrar)) {
139             String msg = "Configuring a AuthenticationStrategy is only applicable when the underlying " +
140                     "Authenticator implementation is a " + AuthenticationListenerRegistrar.class.getName() +
141                     " implementation.  This SecurityManager has been configured with an Authenticator of type " +
142                     this.authenticator.getClass().getName() + ", which does not implement that interface.";
143             throw new IllegalStateException(msg);
144         }
145         ((AuthenticationListenerRegistrar) this.authenticator).setAuthenticationListeners(listeners);
146     }
147 
148     public void add(AuthenticationListener listener) {
149         assertAuthenticatorListenerSupport();
150         ((AuthenticationListenerRegistrar) this.authenticator).add(listener);
151     }
152 
153     public boolean remove(AuthenticationListener listener) {
154         return (this.authenticator instanceof AuthenticationListenerRegistrar) &&
155                 ((AuthenticationListenerRegistrar) this.authenticator).remove(listener);
156     }
157 
158     /**
159      * Ensures that <code>this.authenticator</code> implements the
160      * {@link org.apache.ki.authc.AuthenticationListenerRegistrar AuthenticationListenerRegistrar} interface to ensure
161      * listeners can be registered.
162      */
163     private void assertAuthenticatorListenerSupport() {
164         if (!(this.authenticator instanceof AuthenticationListenerRegistrar)) {
165             String msg = "AuthenticationListener registration failed:  The underlying Authenticator instance of " +
166                     "type [" + this.authenticator.getClass().getName() + "] does not implement the " +
167                     AuthenticationListenerRegistrar.class.getName() + " interface and therefore cannot support " +
168                     "runtime registration of AuthenticationListeners.";
169             throw new IllegalStateException(msg);
170         }
171     }
172 
173     /**
174      * Passes on the {@link #getRealms() realms} to the internal delegate <code>Authenticator</code> instance so
175      * that it may use them during authentication attempts.
176      */
177     protected void afterRealmsSet() {
178         super.afterRealmsSet();
179         if (this.authenticator instanceof ModularRealmAuthenticator) {
180             ((ModularRealmAuthenticator) this.authenticator).setRealms(getRealms());
181         }
182     }
183 
184     /**
185      * Delegates to the wrapped {@link org.apache.ki.authc.Authenticator Authenticator} for authentication.
186      */
187     public AuthenticationInfo authenticate(AuthenticationToken token) throws AuthenticationException {
188         return this.authenticator.authenticate(token);
189     }
190 
191     public void destroy() {
192         LifecycleUtils.destroy(getAuthenticator());
193         this.authenticator = null;
194         super.destroy();
195     }
196 }
