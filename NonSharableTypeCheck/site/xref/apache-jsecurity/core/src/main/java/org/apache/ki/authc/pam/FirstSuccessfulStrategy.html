19  package org.apache.ki.authc.pam;
20  
21  import java.util.Collection;
22  
23  import org.apache.ki.authc.AuthenticationException;
24  import org.apache.ki.authc.AuthenticationInfo;
25  import org.apache.ki.authc.AuthenticationToken;
26  import org.apache.ki.realm.Realm;
27  
28  /**
29   * {@link AuthenticationStrategy} implementation that only accepts the account data from
30   * the first successfully consulted Realm and ignores all subsequent realms.  This is slightly
31   * different behavior than
32   * {@link AtLeastOneSuccessfulStrategy AtLeastOneSuccessfulAuthenticationStrategy},
33   * so please review both to see which one meets your needs better.
34   *
35   * @author Les Hazlewood
36   * @see AtLeastOneSuccessfulStrategy AtLeastOneSuccessfulAuthenticationStrategy
37   * @since 0.9
38   */
39  public class FirstSuccessfulStrategy extends AbstractAuthenticationStrategy {
40  
41      /**
42       * Returns <code>null</code> immediately, relying on this class's {@link #merge merge} implementation to return
43       * only the first <code>info</code> object it encounters, ignoring all subsequent ones.
44       */
45      public AuthenticationInfo beforeAllAttempts(Collection<? extends Realm> realms, AuthenticationToken token) throws AuthenticationException {
46          return null;
47      }
48  
49      /**
50       * Returns the specified <code>aggregate</code> instance if is non null and valid (that is, has principals and they are
51       * not empty) immediately, or, if it is null or not valid, the <code>info</code> argument is returned instead.
52       * <p/>
53       * This logic ensures that the first valid info encountered is the one retained and all subsequent ones are ignored,
54       * since this strategy mandates that only the info from the first successfully authenticated realm be used.
55       */
56      protected AuthenticationInfo merge(AuthenticationInfo info, AuthenticationInfo aggregate) {
57          if (aggregate != null && aggregate.getPrincipals() != null && !aggregate.getPrincipals().isEmpty()) {
58              return aggregate;
59          }
60          return info != null ? info : aggregate;
61      }
62  }
