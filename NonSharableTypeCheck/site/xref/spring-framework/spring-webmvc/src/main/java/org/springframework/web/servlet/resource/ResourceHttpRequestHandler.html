55  /**
56   * {@link HttpRequestHandler} that serves static resources optimized for superior browser performance
57   * (according to the guidelines of Page Speed, YSlow, etc.) by allowing for flexible cache settings
58   * ({@linkplain #setCacheSeconds "cacheSeconds" property}, last-modified support).
59   *
60   * <p>The {@linkplain #setLocations "locations" property} takes a list of Spring {@link Resource}
61   * locations from which static resources are allowed  to be served by this handler. For a given request,
62   * the list of locations will be consulted in order for the presence of the requested resource, and the
63   * first found match will be written to the response, with a HTTP Caching headers
64   * set as configured. The handler also properly evaluates the {@code Last-Modified} header
65   * (if present) so that a {@code 304} status code will be returned as appropriate, avoiding unnecessary
66   * overhead for resources that are already cached by the client. The use of {@code Resource} locations
67   * allows resource requests to easily be mapped to locations other than the web application root.
68   * For example, resources could be served from a classpath location such as
69   * "classpath:/META-INF/public-web-resources/", allowing convenient packaging and serving of resources
70   * such as a JavaScript library from within jar files.
71   *
72   * <p>To ensure that users with a primed browser cache get the latest changes to application-specific
73   * resources upon deployment of new versions of the application, it is recommended that a version
74   * string is used in the URL  mapping pattern that selects this handler. Such patterns can be easily
75   * parameterized using Spring EL. See the reference manual for further examples of this approach.
76   *
77   * <p>For various front-end needs &mdash; such as ensuring that users with a primed browser cache
78   * get the latest changes, or serving variations of resources (e.g., minified versions) &mdash;
79   * {@link org.springframework.web.servlet.resource.ResourceResolver}s can be configured.
80   *
81   * <p>This handler can be configured through use of a
82   * {@link org.springframework.web.servlet.config.annotation.ResourceHandlerRegistry}
83   * or the {@code <mvc:resources/>} XML configuration element.
84   *
85   * @author Keith Donald
86   * @author Jeremy Grelle
87   * @author Juergen Hoeller
88   * @author Arjen Poutsma
89   * @since 3.0.4
90   */
91  public class ResourceHttpRequestHandler extends WebContentGenerator implements HttpRequestHandler, InitializingBean {

166 	@Override
167 	public void afterPropertiesSet() throws Exception {
168 		if (logger.isWarnEnabled() && CollectionUtils.isEmpty(this.locations)) {
169 			logger.warn("Locations list is empty. No resources will be served unless a " +
170 					"custom ResourceResolver is configured as an alternative to PathResourceResolver.");
171 		}
172 		initAllowedLocations();
173 	}
174 
175 	/**
176 	 * Look for a {@link org.springframework.web.servlet.resource.PathResourceResolver}
177 	 * among the {@link #getResourceResolvers() resource resolvers} and configure
178 	 * its {@code "allowedLocations"} to match the value of the
179 	 * {@link #setLocations(java.util.List) locations} property unless the "allowed
180 	 * locations" of the {@code PathResourceResolver} is non-empty.
181 	 */
182 	protected void initAllowedLocations() {
183 		if (CollectionUtils.isEmpty(this.locations)) {
184 			return;
185 		}
186 		for (int i = getResourceResolvers().size()-1; i >= 0; i--) {
187 			if (getResourceResolvers().get(i) instanceof PathResourceResolver) {
188 				PathResourceResolver pathResolver = (PathResourceResolver) getResourceResolvers().get(i);
189 				if (ObjectUtils.isEmpty(pathResolver.getAllowedLocations())) {
190 					pathResolver.setAllowedLocations(getLocations().toArray(new Resource[getLocations().size()]));
191 				}
192 				break;
193 			}
194 		}
195 	}

197 	/**
198 	 * Processes a resource request.
199 	 * <p>Checks for the existence of the requested resource in the configured list of locations.
200 	 * If the resource does not exist, a {@code 404} response will be returned to the client.
201 	 * If the resource exists, the request will be checked for the presence of the
202 	 * {@code Last-Modified} header, and its value will be compared against the last-modified
203 	 * timestamp of the given resource, returning a {@code 304} status code if the
204 	 * {@code Last-Modified} value  is greater. If the resource is newer than the
205 	 * {@code Last-Modified} value, or the header is not present, the content resource
206 	 * of the resource will be written to the response with caching headers
207 	 * set to expire one year in the future.
208 	 */
209 	@Override
210 	public void handleRequest(HttpServletRequest request, HttpServletResponse response)
211 			throws ServletException, IOException {
212 
213 		checkAndPrepare(request, response);
214 
215 		// check whether a matching resource exists
216 		Resource resource = getResource(request);
217 		if (resource == null) {
218 			logger.trace("No matching resource found - returning 404");
219 			response.sendError(HttpServletResponse.SC_NOT_FOUND);
220 			return;
221 		}
222 
223 		// header phase
224 		if (new ServletWebRequest(request, response).checkNotModified(resource.lastModified())) {
225 			logger.trace("Resource not modified - returning 304");
226 			return;
227 		}
228 
229 		// check the resource's media type
230 		MediaType mediaType = getMediaType(resource);
231 		if (mediaType != null) {
232 			if (logger.isTraceEnabled()) {
233 				logger.trace("Determined media type '" + mediaType + "' for " + resource);
234 			}
235 		}
236 		else {
237 			if (logger.isTraceEnabled()) {
238 				logger.trace("No media type found for " + resource + " - not sending a content-type header");
239 			}
240 		}
241 
242 		// content phase
243 		if (METHOD_HEAD.equals(request.getMethod())) {
244 			setHeaders(response, resource, mediaType);
245 			logger.trace("HEAD request - skipping content");
246 			return;
247 		}
248 
249 		if (request.getHeader(HttpHeaders.RANGE) == null) {
250 			setHeaders(response, resource, mediaType);
251 			writeContent(response, resource);
252 		}
253 		else {
254 			writePartialContent(request, response, resource, mediaType);
255 		}
256 	}

258 	protected Resource getResource(HttpServletRequest request) throws IOException {
259 		String path = (String) request.getAttribute(HandlerMapping.PATH_WITHIN_HANDLER_MAPPING_ATTRIBUTE);
260 		if (path == null) {
261 			throw new IllegalStateException("Required request attribute '" +
262 					HandlerMapping.PATH_WITHIN_HANDLER_MAPPING_ATTRIBUTE + "' is not set");
263 		}
264 		path = processPath(path);
265 		if (!StringUtils.hasText(path) || isInvalidPath(path)) {
266 			if (logger.isTraceEnabled()) {
267 				logger.trace("Ignoring invalid resource path [" + path + "]");
268 			}
269 			return null;
270 		}
271 		if (path.contains("%")) {
272 			try {
273 				// Use URLDecoder (vs UriUtils) to preserve potentially decoded UTF-8 chars
274 				if (isInvalidPath(URLDecoder.decode(path, "UTF-8"))) {
275 					if (logger.isTraceEnabled()) {
276 						logger.trace("Ignoring invalid resource path with escape sequences [" + path + "].");
277 					}
278 					return null;
279 				}
280 			}
281 			catch (IllegalArgumentException ex) {
282 				// ignore
283 			}
284 		}
285 		ResourceResolverChain resolveChain = new DefaultResourceResolverChain(getResourceResolvers());
286 		Resource resource = resolveChain.resolveResource(request, path, getLocations());
287 		if (resource == null || getResourceTransformers().isEmpty()) {
288 			return resource;
289 		}
290 		ResourceTransformerChain transformChain =
291 				new DefaultResourceTransformerChain(resolveChain, getResourceTransformers());
292 		resource = transformChain.transform(request, resource);
293 		return resource;
294 	}

296 	/**
297 	 * Process the given resource path to be used.
298 	 * <p>The default implementation replaces any combination of leading '/' and
299 	 * control characters (00-1F and 7F) with a single "/" or "". For example
300 	 * {@code "  // /// ////  foo/bar"} becomes {@code "/foo/bar"}.
301 	 * @since 3.2.12
302 	 */
303 	protected String processPath(String path) {
304 		boolean slash = false;
305 		for (int i = 0; i < path.length(); i++) {
306 			if (path.charAt(i) == '/') {
307 				slash = true;
308 			}
309 			else if (path.charAt(i) > ' ' && path.charAt(i) != 127) {
310 				if (i == 0 || (i == 1 && slash)) {
311 					return path;
312 				}
313 				path = slash ? "/" + path.substring(i) : path.substring(i);
314 				if (logger.isTraceEnabled()) {
315 					logger.trace("Path after trimming leading '/' and control characters: " + path);
316 				}
317 				return path;
318 			}
319 		}
320 		return (slash ? "/" : "");
321 	}

323 	/**
324 	 * Identifies invalid resource paths. By default rejects:
325 	 * <ul>
326 	 * <li>Paths that contain "WEB-INF" or "META-INF"
327 	 * <li>Paths that contain "../" after a call to
328 	 * {@link org.springframework.util.StringUtils#cleanPath}.
329 	 * <li>Paths that represent a {@link org.springframework.util.ResourceUtils#isUrl
330 	 * valid URL} or would represent one after the leading slash is removed.
331 	 * </ul>
332 	 * <p><strong>Note:</strong> this method assumes that leading, duplicate '/'
333 	 * or control characters (e.g. white space) have been trimmed so that the
334 	 * path starts predictably with a single '/' or does not have one.
335 	 * @param path the path to validate
336 	 * @return {@code true} if the path is invalid, {@code false} otherwise
337 	 */
338 	protected boolean isInvalidPath(String path) {
339 		if (logger.isTraceEnabled()) {
340 			logger.trace("Applying \"invalid path\" checks to path: " + path);
341 		}
342 		if (path.contains("WEB-INF") || path.contains("META-INF")) {
343 			if (logger.isTraceEnabled()) {
344 				logger.trace("Path contains \"WEB-INF\" or \"META-INF\".");
345 			}
346 			return true;
347 		}
348 		if (path.contains(":/")) {
349 			String relativePath = (path.charAt(0) == '/' ? path.substring(1) : path);
350 			if (ResourceUtils.isUrl(relativePath) || relativePath.startsWith("url:")) {
351 				if (logger.isTraceEnabled()) {
352 					logger.trace("Path represents URL or has \"url:\" prefix.");
353 				}
354 				return true;
355 			}
356 		}
357 		if (path.contains("../")) {
358 			path = StringUtils.cleanPath(path);
359 			if (path.contains("../")) {
360 				if (logger.isTraceEnabled()) {
361 					logger.trace("Path contains \"../\" after call to StringUtils#cleanPath.");
362 				}
363 				return true;
364 			}
365 		}
366 		return false;
367 	}

369 	/**
370 	 * Determine an appropriate media type for the given resource.
371 	 * @param resource the resource to check
372 	 * @return the corresponding media type, or {@code null} if none found
373 	 */
374 	protected MediaType getMediaType(Resource resource) {
375 		MediaType mediaType = null;
376 		String mimeType = getServletContext().getMimeType(resource.getFilename());
377 		if (StringUtils.hasText(mimeType)) {
378 			mediaType = MediaType.parseMediaType(mimeType);
379 		}
380 		if (jafPresent && (mediaType == null || MediaType.APPLICATION_OCTET_STREAM.equals(mediaType))) {
381 			MediaType jafMediaType = ActivationMediaTypeFactory.getMediaType(resource.getFilename());
382 			if (jafMediaType != null && !MediaType.APPLICATION_OCTET_STREAM.equals(jafMediaType)) {
383 				mediaType = jafMediaType;
384 			}
385 		}
386 		return mediaType;
387 	}

389 	/**
390 	 * Set headers on the given servlet response.
391 	 * Called for GET requests as well as HEAD requests.
392 	 * @param response current servlet response
393 	 * @param resource the identified resource (never {@code null})
394 	 * @param mediaType the resource's media type (never {@code null})
395 	 * @throws IOException in case of errors while setting the headers
396 	 */
397 	protected void setHeaders(HttpServletResponse response, Resource resource, MediaType mediaType) throws IOException {
398 		long length = resource.contentLength();
399 		if (length > Integer.MAX_VALUE) {
400 			throw new IOException("Resource content too long (beyond Integer.MAX_VALUE): " + resource);
401 		}
402 		response.setContentLength((int) length);
403 
404 		if (mediaType != null) {
405 			response.setContentType(mediaType.toString());
406 		}
407 
408 		if (resource instanceof EncodedResource) {
409 			response.setHeader(CONTENT_ENCODING, ((EncodedResource) resource).getContentEncoding());
410 		}
411 
412 		response.setHeader(HttpHeaders.ACCEPT_RANGES, "bytes");
413 	}

415 	/**
416 	 * Write the actual content out to the given servlet response,
417 	 * streaming the resource's content.
418 	 * @param response current servlet response
419 	 * @param resource the identified resource (never {@code null})
420 	 * @throws IOException in case of errors while writing the content
421 	 */
422 	protected void writeContent(HttpServletResponse response, Resource resource) throws IOException {
423 		InputStream in = resource.getInputStream();
424 		try {
425 			StreamUtils.copy(in, response.getOutputStream());
426 		}
427 		finally {
428 			try {
429 				in.close();
430 			}
431 			catch (IOException ex) {
432 				// ignore
433 			}
434 		}
435 	}

437 	/**
438 	 * Write parts of the resource as indicated by the request {@code Range} header.
439 	 * @param request current servlet request
440 	 * @param response current servlet response
441 	 * @param resource the identified resource (never {@code null})
442 	 * @param contentType the content type
443 	 * @throws IOException in case of errors while writing the content
444 	 */
445 	protected void writePartialContent(HttpServletRequest request, HttpServletResponse response,
446 			Resource resource, MediaType contentType) throws IOException {
447 
448 		long length = resource.contentLength();
449 
450 		List<HttpRange> ranges;
451 		try {
452 			HttpHeaders headers = new ServletServerHttpRequest(request).getHeaders();
453 			ranges = headers.getRange();
454 		}
455 		catch (IllegalArgumentException ex) {
456 			response.addHeader("Content-Range", "bytes */" + length);
457 			response.sendError(HttpServletResponse.SC_REQUESTED_RANGE_NOT_SATISFIABLE);
458             return;
459 		}
460 
461 		response.setStatus(HttpServletResponse.SC_PARTIAL_CONTENT);
462 
463 		if (ranges.size() == 1) {
464 			HttpRange range = ranges.get(0);
465 
466 			long start = range.getRangeStart(length);
467 			long end = range.getRangeEnd(length);
468 			long rangeLength = end - start + 1;
469 
470 			setHeaders(response, resource, contentType);
471 			response.addHeader("Content-Range", "bytes " + start + "-" + end + "/" + length);
472             response.setContentLength((int) rangeLength);
473 
474 			InputStream in = resource.getInputStream();
475 			try {
476 				copyRange(in, response.getOutputStream(), start, end);
477 			}
478 			finally {
479 				try {
480 					in.close();
481 				}
482 				catch (IOException ex) {
483 					// ignore
484 				}
485 			}
486 		}
487 		else {
488 			String boundaryString = MimeTypeUtils.generateMultipartBoundaryString();
489 			response.setContentType("multipart/byteranges; boundary=" + boundaryString);
490 
491 			ServletOutputStream out = response.getOutputStream();
492 
493 			for (HttpRange range : ranges) {
494 				long start = range.getRangeStart(length);
495 				long end = range.getRangeEnd(length);
496 
497 				InputStream in = resource.getInputStream();
498 
499                 // Writing MIME header.
500                 out.println();
501                 out.println("--" + boundaryString);
502                 if (contentType != null) {
503 	                out.println("Content-Type: " + contentType);
504                 }
505                 out.println("Content-Range: bytes " + start + "-" + end + "/" + length);
506                 out.println();
507 
508                 // Printing content
509                 copyRange(in, out, start, end);
510 			}
511 			out.println();
512             out.print("--" + boundaryString + "--");
513 		}
514 	}
515 
516 	private void copyRange(InputStream in, OutputStream out, long start, long end) throws IOException {
517 
518 		long skipped = in.skip(start);
519 
520 		if (skipped < start) {
521 			throw new IOException("Skipped only " + skipped + " bytes out of " + start + " required.");
522 		}
523 
524 		long bytesToCopy = end - start + 1;
525 
526 		byte buffer[] = new byte[StreamUtils.BUFFER_SIZE];
527 		while (bytesToCopy > 0) {
528 			int bytesRead = in.read(buffer);
529 			if (bytesRead <= bytesToCopy) {
530 				out.write(buffer, 0, bytesRead);
531 				bytesToCopy -= bytesRead;
532 			}
533 			else {
534 				out.write(buffer, 0, (int) bytesToCopy);
535 				bytesToCopy = 0;
536 			}
537 			if (bytesRead < buffer.length) {
538 				break;
539 			}
540 		}
541 	}

552 	 * Inner class to avoid a hard-coded JAF dependency.
553 	 */
554 	private static class ActivationMediaTypeFactory {
555 
556 		private static final FileTypeMap fileTypeMap;
557 
558 		static {
559 			fileTypeMap = loadFileTypeMapFromContextSupportModule();
560 		}
561 
562 		private static FileTypeMap loadFileTypeMapFromContextSupportModule() {
563 			// See if we can find the extended mime.types from the context-support module...
564 			Resource mappingLocation = new ClassPathResource("org/springframework/mail/javamail/mime.types");
565 			if (mappingLocation.exists()) {
566 				InputStream inputStream = null;
567 				try {
568 					inputStream = mappingLocation.getInputStream();
569 					return new MimetypesFileTypeMap(inputStream);
570 				}
571 				catch (IOException ex) {
572 					// ignore
573 				}
574 				finally {
575 					if (inputStream != null) {
576 						try {
577 							inputStream.close();
578 						}
579 						catch (IOException ex) {
580 							// ignore
581 						}
582 					}
583 				}
584 			}
585 			return FileTypeMap.getDefaultFileTypeMap();
586 		}

588 		public static MediaType getMediaType(String filename) {
589 			String mediaType = fileTypeMap.getContentType(filename);
590 			return (StringUtils.hasText(mediaType) ? MediaType.parseMediaType(mediaType) : null);
