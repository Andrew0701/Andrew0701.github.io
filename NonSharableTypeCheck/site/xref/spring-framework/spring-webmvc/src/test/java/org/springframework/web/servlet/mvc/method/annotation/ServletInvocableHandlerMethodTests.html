170 	private void wrapConcurrentResult_ResponseBody(Object handler) throws Exception {
171 		List<HttpMessageConverter<?>> converters = new ArrayList<HttpMessageConverter<?>>();
172 		converters.add(new StringHttpMessageConverter());
173 		this.returnValueHandlers.addHandler(new RequestResponseBodyMethodProcessor(converters));

175 		handlerMethod = handlerMethod.wrapConcurrentResult("bar");
176 		handlerMethod.invokeAndHandle(this.webRequest, this.mavContainer);
177 
178 		assertEquals("bar", this.response.getContentAsString());

264 	private ServletInvocableHandlerMethod getHandlerMethod(Object controller,
265 			String methodName, Class<?>... argTypes) throws NoSuchMethodException {
266 
267 		Method method = controller.getClass().getDeclaredMethod(methodName, argTypes);
268 		ServletInvocableHandlerMethod handlerMethod = new ServletInvocableHandlerMethod(controller, method);
269 		handlerMethod.setHandlerMethodArgumentResolvers(this.argumentResolvers);
270 		handlerMethod.setHandlerMethodReturnValueHandlers(this.returnValueHandlers);
271 		return handlerMethod;
272 	}
