45      @Test
46      public void testAliasCrudRouting() throws Exception {
47          createIndex("test");
48          ensureGreen();
49          assertAcked(admin().indices().prepareAliases().addAliasAction(newAddAliasAction("test", "alias0").routing("0")));
50  
51          logger.info("--> indexing with id [1], and routing [0] using alias");
52          client().prepareIndex("alias0", "type1", "1").setSource("field", "value1").setRefresh(true).execute().actionGet();
53          logger.info("--> verifying get with no routing, should not find anything");
54          for (int i = 0; i < 5; i++) {
55              assertThat(client().prepareGet("test", "type1", "1").execute().actionGet().isExists(), equalTo(false));
56          }
57          logger.info("--> verifying get with routing, should find");
58          for (int i = 0; i < 5; i++) {
59              assertThat(client().prepareGet("test", "type1", "1").setRouting("0").execute().actionGet().isExists(), equalTo(true));
60          }
61  
62          logger.info("--> verifying get with routing alias, should find");
63          for (int i = 0; i < 5; i++) {
64              assertThat(client().prepareGet("alias0", "type1", "1").execute().actionGet().isExists(), equalTo(true));
65          }
66  
67          logger.info("--> updating with id [1] and routing through alias");
68          client().prepareUpdate("alias0", "type1", "1")
69                  .setUpsert(XContentFactory.jsonBuilder().startObject().field("field", 1).endObject())
70                  .setScript("ctx._source.field = 'value2'", ScriptService.ScriptType.INLINE)
71                  .execute().actionGet();
72          for (int i = 0; i < 5; i++) {
73              assertThat(client().prepareGet("alias0", "type1", "1").execute().actionGet().isExists(), equalTo(true));
74              assertThat(client().prepareGet("alias0", "type1", "1").execute().actionGet().getSourceAsMap().get("field").toString(), equalTo("value2"));
75          }
76  
77  
78          logger.info("--> deleting with no routing, should not delete anything");
79          client().prepareDelete("test", "type1", "1").setRefresh(true).execute().actionGet();
80          for (int i = 0; i < 5; i++) {
81              assertThat(client().prepareGet("test", "type1", "1").execute().actionGet().isExists(), equalTo(false));
82              assertThat(client().prepareGet("test", "type1", "1").setRouting("0").execute().actionGet().isExists(), equalTo(true));
83              assertThat(client().prepareGet("alias0", "type1", "1").execute().actionGet().isExists(), equalTo(true));
84          }
85  
86          logger.info("--> deleting with routing alias, should delete");
87          client().prepareDelete("alias0", "type1", "1").setRefresh(true).execute().actionGet();
88          for (int i = 0; i < 5; i++) {
89              assertThat(client().prepareGet("test", "type1", "1").execute().actionGet().isExists(), equalTo(false));
90              assertThat(client().prepareGet("test", "type1", "1").setRouting("0").execute().actionGet().isExists(), equalTo(false));
91              assertThat(client().prepareGet("alias0", "type1", "1").execute().actionGet().isExists(), equalTo(false));
92          }
93  
94          logger.info("--> indexing with id [1], and routing [0] using alias");
95          client().prepareIndex("alias0", "type1", "1").setSource("field", "value1").setRefresh(true).execute().actionGet();
96          logger.info("--> verifying get with no routing, should not find anything");
97          for (int i = 0; i < 5; i++) {
98              assertThat(client().prepareGet("test", "type1", "1").execute().actionGet().isExists(), equalTo(false));
99          }
100         logger.info("--> verifying get with routing, should find");
101         for (int i = 0; i < 5; i++) {
102             assertThat(client().prepareGet("test", "type1", "1").setRouting("0").execute().actionGet().isExists(), equalTo(true));
103             assertThat(client().prepareGet("alias0", "type1", "1").execute().actionGet().isExists(), equalTo(true));
104         }
105 
106         logger.info("--> deleting_by_query with 1 as routing, should not delete anything");
107         client().prepareDeleteByQuery().setQuery(matchAllQuery()).setRouting("1").execute().actionGet();
108         refresh();
109         for (int i = 0; i < 5; i++) {
110             assertThat(client().prepareGet("test", "type1", "1").execute().actionGet().isExists(), equalTo(false));
111             assertThat(client().prepareGet("test", "type1", "1").setRouting("0").execute().actionGet().isExists(), equalTo(true));
112             assertThat(client().prepareGet("alias0", "type1", "1").execute().actionGet().isExists(), equalTo(true));
113         }
114 
115         logger.info("--> deleting_by_query with alias0, should delete");
116         client().prepareDeleteByQuery("alias0").setQuery(matchAllQuery()).execute().actionGet();
117         refresh();
118         for (int i = 0; i < 5; i++) {
119             assertThat(client().prepareGet("test", "type1", "1").execute().actionGet().isExists(), equalTo(false));
120             assertThat(client().prepareGet("test", "type1", "1").setRouting("0").execute().actionGet().isExists(), equalTo(false));
121             assertThat(client().prepareGet("alias0", "type1", "1").execute().actionGet().isExists(), equalTo(false));
122         }

275     /*
276     See https://github.com/elasticsearch/elasticsearch/issues/2682
277     Searching on more than one index, if one of those is an alias with configured routing, the shards that belonged
278     to the other indices (without routing) were not taken into account in PlainOperationRouting#searchShards.
279     That affected the number of shards that we executed the search on, thus some documents were missing in the search results.
280      */
281     @Test
282     public void testAliasSearchRoutingWithConcreteAndAliasedIndices_issue2682() throws Exception {
283         createIndex("index", "index_2");
284         ensureGreen();
285         assertAcked(admin().indices().prepareAliases()
286                 .addAliasAction(newAddAliasAction("index", "index_1").routing("1")));
287 
288         logger.info("--> indexing on index_1 which is an alias for index with routing [1]");
289         client().prepareIndex("index_1", "type1", "1").setSource("field", "value1").setRefresh(true).execute().actionGet();
290         logger.info("--> indexing on index_2 which is a concrete index");
291         client().prepareIndex("index_2", "type2", "2").setSource("field", "value2").setRefresh(true).execute().actionGet();
292 
293 
294         logger.info("--> search all on index_* should find two");
295         for (int i = 0; i < 5; i++) {
296             assertThat(client().prepareSearch("index_*").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getHits().totalHits(), equalTo(2l));
297         }
298     }

300     /*
301     See https://github.com/elasticsearch/elasticsearch/pull/3268
302     Searching on more than one index, if one of those is an alias with configured routing, the shards that belonged
303     to the other indices (without routing) were not taken into account in PlainOperationRouting#searchShardsCount.
304     That could cause returning 1, which led to forcing the QUERY_AND_FETCH mode.
305     As a result, (size * number of hit shards) results were returned and no reduce phase was taking place.
306      */
307     @Test
308     public void testAliasSearchRoutingWithConcreteAndAliasedIndices_issue3268() throws Exception {
309         createIndex("index", "index_2");
310         ensureGreen();
311         assertAcked(admin().indices().prepareAliases()
312                 .addAliasAction(newAddAliasAction("index", "index_1").routing("1")));
313 
314         logger.info("--> indexing on index_1 which is an alias for index with routing [1]");
315         client().prepareIndex("index_1", "type1", "1").setSource("field", "value1").setRefresh(true).execute().actionGet();
316         logger.info("--> indexing on index_2 which is a concrete index");
317         client().prepareIndex("index_2", "type2", "2").setSource("field", "value2").setRefresh(true).execute().actionGet();
318 
319         SearchResponse searchResponse = client().prepareSearch("index_*").setSearchType(SearchType.QUERY_THEN_FETCH).setSize(1).setQuery(QueryBuilders.matchAllQuery()).execute().actionGet();
320 
321         logger.info("--> search all on index_* should find two");
322         assertThat(searchResponse.getHits().totalHits(), equalTo(2L));
323         //Let's make sure that, even though 2 docs are available, only one is returned according to the size we set in the request
324         //Therefore the reduce phase has taken place, which proves that the QUERY_AND_FETCH search type wasn't erroneously forced.
325         assertThat(searchResponse.getHits().getHits().length, equalTo(1));
326     }

328     @Test

330         createIndex("test");
331         ensureGreen();
332         logger.info("--> creating alias with routing [3]");
333         assertAcked(admin().indices().prepareAliases()
334                 .addAliasAction(newAddAliasAction("test", "alias").routing("3")));
335 
336         logger.info("--> indexing with id [0], and routing [3]");
337         client().prepareIndex("alias", "type1", "0").setSource("field", "value1").setRefresh(true).execute().actionGet();
338         logger.info("--> verifying get with no routing, should not find anything");
339 
340         logger.info("--> verifying get and search with routing, should find");
341         for (int i = 0; i < 5; i++) {
342             assertThat(client().prepareGet("test", "type1", "0").setRouting("3").execute().actionGet().isExists(), equalTo(true));
343             assertThat(client().prepareSearch("alias").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getHits().totalHits(), equalTo(1l));
344             assertThat(client().prepareCount("alias").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount(), equalTo(1l));
345         }
346 
347         logger.info("--> creating alias with routing [4]");
348         assertAcked(admin().indices().prepareAliases()
349                 .addAliasAction(newAddAliasAction("test", "alias").routing("4")));
350 
351         logger.info("--> verifying search with wrong routing should not find");
352         for (int i = 0; i < 5; i++) {
353             assertThat(client().prepareSearch("alias").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getHits().totalHits(), equalTo(0l));
354             assertThat(client().prepareCount("alias").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount(), equalTo(0l));
355         }
356 
357         logger.info("--> creating alias with search routing [3,4] and index routing 4");
358         assertAcked(client().admin().indices().prepareAliases()
359                 .addAliasAction(newAddAliasAction("test", "alias").searchRouting("3,4").indexRouting("4")));
360 
361         logger.info("--> indexing with id [1], and routing [4]");
362         client().prepareIndex("alias", "type1", "1").setSource("field", "value2").setRefresh(true).execute().actionGet();
363         logger.info("--> verifying get with no routing, should not find anything");
364 
365         logger.info("--> verifying get and search with routing, should find");
366         for (int i = 0; i < 5; i++) {
367             assertThat(client().prepareGet("test", "type1", "0").setRouting("3").execute().actionGet().isExists(), equalTo(true));
368             assertThat(client().prepareGet("test", "type1", "1").setRouting("4").execute().actionGet().isExists(), equalTo(true));
369             assertThat(client().prepareSearch("alias").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getHits().totalHits(), equalTo(2l));
370             assertThat(client().prepareCount("alias").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount(), equalTo(2l));
371         }
